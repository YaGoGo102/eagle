const mineflayer = require('mineflayer'); const { WebSocketServer } = require('ws'); const wss = new WebSocketServer({ port: 8080 }); wss.on('connection', (ws) => { let myBot = null; ws.on('message', (message) => { try { const data = JSON.parse(message); if (data.type === 'login' && !myBot) { myBot = mineflayer.createBot({ host: 'localhost', port: 25565, username: data.username, version: '1.20.1', skipValidation: true }); myBot.on('spawn', () => ws.send(JSON.stringify({ type: 'login_success' }))); myBot.on('error', (err) => console.log(err)); } if (data.type === 'move' && myBot) { myBot.setControlState('forward', data.forward); myBot.setControlState('back', data.back); myBot.setControlState('left', data.left); myBot.setControlState('right', data.right); myBot.setControlState('jump', data.jump); } if (data.type === 'look' && myBot) { myBot.look(data.yaw, data.pitch, true); } } catch (e) {} }); }); console.log('🚀 安定版サーバー起動中...');
